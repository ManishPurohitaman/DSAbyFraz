class Solution {
public:
    int maxProfit(vector<int>& prices)
    {
        int profit=0;
        int min_price=1e5;
        int n=prices.size();
        
        for(int i=0; i<n; i++)
        {
                profit=max(profit, prices[i]-min_price);
                min_price=min(min_price, prices[i]);
        }
        
        return profit;
    }
};
